## 3.2 BaseTool 基类

**文件位置**：`app/tool/base.py`

### 3.2.1 基类定义

```python
class BaseTool(ABC, BaseModel):
    """工具抽象基类"""

    name: str                                    # 工具名称
    description: str                            # 工具描述
    parameters: Optional[dict] = None           # 参数模式

    class Config:
        arbitrary_types_allowed = True

    async def __call__(self, **kwargs) -> Any:
        """执行工具"""
        return await self.execute(**kwargs)

    @abstractmethod
    async def execute(self, **kwargs) -> Any:
        """工具执行逻辑"""
        pass

    def to_param(self) -> Dict:
        """转换为函数调用格式"""
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": self.parameters,
            },
        }
```

## 3.3 ToolResult 数据结构

```python
class ToolResult(BaseModel):
    """工具执行结果"""

    output: Any = None                    # 正常输出
    error: Optional[str] = None          # 错误信息
    base64_image: Optional[str] = None   # 截图数据
    system: Optional[str] = None         # 系统信息

    def __bool__(self):
        """判断结果是否有内容"""
        return any(getattr(self, field) for field in self.__fields__)

    def __str__(self):
        """字符串表示"""
        return f"Error: {self.error}" if self.error else self.output

    def __add__(self, other: "ToolResult") -> "ToolResult":
        """合并多个结果"""
        return ToolResult(
            output=combine_fields(self.output, other.output),
            error=combine_fields(self.error, other.error),
            base64_image=self.base64_image or other.base64_image,
            system=combine_fields(self.system, other.system),
        )
```

## 3.4 ToolCollection 工具集合

**文件位置**：`app/tool/tool_collection.py`

```python
class ToolCollection:
    """工具集合管理器"""

    def __init__(self, *tools: BaseTool):
        self.tools = tools
        self.tool_map = {tool.name: tool for tool in tools}

    def to_params(self) -> List[Dict[str, Any]]:
        """转换为 LLM 工具参数格式"""
        return [tool.to_param() for tool in self.tools]

    async def execute(
        self,
        name: str,
        tool_input: Dict[str, Any] = None
    ) -> ToolResult:
        """执行指定工具"""
        tool = self.tool_map.get(name)
        if not tool:
            return ToolFailure(error=f"Tool {name} is invalid")
        try:
            result = await tool(**tool_input)
            return result
        except ToolError as e:
            return ToolFailure(error=e.message)

    def add_tool(self, tool: BaseTool):
        """添加工具（不重复添加）"""
        if tool.name in self.tool_map:
            logger.warning(f"Tool {tool.name} already exists")
            return self
        self.tools += (tool,)
        self.tool_map[tool.name] = tool
        return self

    def add_tools(self, *tools: BaseTool):
        """批量添加工具"""
        for tool in tools:
            self.add_tool(tool)
        return self
```

### 3.5.1 Bash 工具

**文件位置**：`app/tool/bash.py`

执行本地 Bash 命令的会话式工具：

```python
class Bash(BaseTool):
    name: str = "bash"
    description: str = "Execute bash commands"

    async def execute(
        self,
        command: str,
        timeout: int = 30
    ) -> ToolResult:
        """执行 Bash 命令"""
        # 创建会话（如果不存在）
        if not self.shell:
            self.shell = await create_spawn_shell()

        # 执行命令
        result = await self.shell.run_command(command, timeout)
        return ToolResult(output=result)
```

### 3.5.3 StrReplaceEditor 文件编辑器

**文件位置**：`app/tool/str_replace_editor.py`

```python
class StrReplaceEditor(BaseTool):
    name: str = "str_replace_editor"
    description: str = "View, create, and edit files"

    # 支持的动作
    ACTIONS = ["view", "create", "str_replace", "insert", "undo_edit"]

    async def execute(
        self,
        command: str,
        file_path: str,
        **kwargs
    ) -> ToolResult:
        """文件编辑操作"""
        if command == "view":
            return await self._view_file(file_path)
        elif command == "create":
            return await self._create_file(file_path, kwargs.get("text"))
        elif command == "str_replace":
            return await self._str_replace(
                file_path,
                kwargs.get("old_str"),
                kwargs.get("new_str")
            )
        elif command == "insert":
            return await self._insert_text(
                file_path,
                kwargs.get("text"),
                kwargs.get("position")
            )
```

### 3.5.4 WebSearch 网络搜索

**文件位置**：`app/tool/web_search.py`

```python
class WebSearch(BaseTool):
    name: str = "web_search"
    description: str = "Search the web for information"

    # 支持多个搜索引擎
    ENGINES = {
        "google": GoogleSearch,
        "bing": BingSearch,
        "baidu": BaiduSearch,
        "duckduckgo": DuckDuckGoSearch,
    }

    async def execute(
        self,
        query: str,
        engine: str = "Google",
        num_results: int = 10
    ) -> ToolResult:
        """执行网络搜索"""
        search_engine = self.ENGINES.get(engine.lower())
        if not search_engine:
            return ToolResult(error=f"Unknown engine: {engine}")

        try:
            results = await search_engine.search(query, num_results)
            return ToolResult(output=str(results))
        except Exception as e:
            return ToolResult(error=str(e))
```

## 3.6 MCP 客户端工具

**文件位置**：`app/tool/mcp.py`

MCP 客户端工具用于代理远程 MCP 服务器的工具：

```python
class MCPClientTool(BaseTool):
    """MCP 服务器工具代理"""

    session: Optional[ClientSession] = None
    server_id: str = ""
    original_name: str = ""

    async def execute(self, **kwargs) -> ToolResult:
        """通过 MCP 服务器执行工具"""
        if not self.session:
            return ToolResult(error="Not connected to MCP server")

        try:
            result = await self.session.call_tool(
                self.original_name,
                kwargs
            )
            content_str = ", ".join(
                item.text for item in result.content
                if isinstance(item, TextContent)
            )
            return ToolResult(output=content_str)
        except Exception as e:
            return ToolResult(error=str(e))
```

## 3.7 工具使用流程

```
Agent.think()
    │
    ▼
┌─────────────────────────────────┐
│  LLM.ask_tool()                 │
│  - 发送 messages + tools        │
└────────┬────────────────────────┘
         │
         ▼
┌─────────────────────────────────┐
│  解析 LLM 返回的 tool_calls      │
└────────┬────────────────────────┘
         │
         ▼
┌─────────────────────────────────┐
│  ToolCallAgent.execute_tool()   │
│  - 查找工具: tool_map[name]     │
│  - 解析参数: json.loads(args)   │
│  - 执行工具: tool.execute(**)   │
└────────┬────────────────────────┘
         │
         ▼
┌─────────────────────────────────┐
│  ToolResult / ToolFailure       │
│  - output / error               │
│  - 添加到 memory                │
└─────────────────────────────────┘
```

## 3.9 自定义工具示例

```python
from app.tool.base import BaseTool, ToolResult

class MyCustomTool(BaseTool):
    name: str = "my_custom_tool"
    description: str = "A custom tool for specific tasks"
    parameters: dict = {
        "type": "object",
        "properties": {
            "input_data": {
                "type": "string",
                "description": "Input data for processing"
            }
        },
        "required": ["input_data"]
    }

    async def execute(self, input_data: str, **kwargs) -> ToolResult:
        """执行自定义逻辑"""
        try:
            # 自定义处理逻辑
            result = process_data(input_data)
            return ToolResult(output=result)
        except Exception as e:
            return ToolResult(error=str(e))

# 使用自定义工具
agent.available_tools.add_tool(MyCustomTool())
```
