### 8.2.2 BaseSandboxClient 抽象类

```python
class BaseSandboxClient(ABC):
    """沙箱客户端基类"""

    @abstractmethod
    async def create(
        self,
        config: Optional[SandboxSettings] = None,
        volume_bindings: Optional[Dict[str, str]] = None,
    ) -> None:
        """创建沙箱"""
        pass

    @abstractmethod
    async def run_command(self, command: str, timeout: Optional[int] = None) -> str:
        """执行命令"""
        pass

    @abstractmethod
    async def cleanup(self) -> None:
        """清理资源"""
        pass
```

## 8.3 LocalSandboxClient 实现

**文件位置**：`app/sandbox/client.py`

```python
class LocalSandboxClient(BaseSandboxClient):
    """本地 Docker 沙箱客户端"""

    def __init__(self):
        self.sandbox: Optional[DockerSandbox] = None

    async def create(
        self,
        config: Optional[SandboxSettings] = None,
        volume_bindings: Optional[Dict[str, str]] = None,
    ) -> None:
        """创建 Docker 沙箱"""
        self.sandbox = DockerSandbox(config, volume_bindings)
        await self.sandbox.create()

    async def run_command(self, command: str, timeout: Optional[int] = None) -> str:
        """在沙箱中执行命令"""
        if not self.sandbox:
            raise RuntimeError("Sandbox not initialized")
        return await self.sandbox.run_command(command, timeout)

    async def copy_from(self, container_path: str, local_path: str) -> None:
        """从容器复制文件"""
        if not self.sandbox:
            raise RuntimeError("Sandbox not initialized")
        await self.sandbox.copy_from(container_path, local_path)

    async def copy_to(self, local_path: str, container_path: str) -> None:
        """上传文件到容器"""
        if not self.sandbox:
            raise RuntimeError("Sandbox not initialized")
        await self.sandbox.copy_to(local_path, container_path)

    async def read_file(self, path: str) -> str:
        """读取容器文件"""
        if not self.sandbox:
            raise RuntimeError("Sandbox not initialized")
        return await self.sandbox.read_file(path)

    async def write_file(self, path: str, content: str) -> None:
        """写入容器文件"""
        if not self.sandbox:
            raise RuntimeError("Sandbox not initialized")
        await self.sandbox.write_file(path, content)

    async def cleanup(self) -> None:
        """清理沙箱"""
        if self.sandbox:
            await self.sandbox.cleanup()
            self.sandbox = None
```

## 8.4 DockerSandbox 核心实现

**文件位置**：`app/sandbox/core/sandbox.py`

### 8.4.1 容器创建

```python
class DockerSandbox:
    """Docker 容器沙箱"""

    def __init__(
        self,
        config: Optional[SandboxSettings] = None,
        volume_bindings: Optional[Dict[str, str]] = None,
    ):
        self.config = config or SandboxSettings()
        self.volume_bindings = volume_bindings or {}
        self.container: Optional[Container] = None
        self.docker_client: Optional[docker.DockerClient] = None

    async def create(self) -> None:
        """创建 Docker 容器"""
        # 初始化 Docker 客户端
        self.docker_client = docker.from_env()

        # 构建卷挂载
        binds = {
            host_path: {"bind": container_path, "mode": "rw"}
            for host_path, container_path in self.volume_bindings.items()
        }

        # 创建容器
        self.container = self.docker_client.containers.run(
            image=self.config.image,
            detach=True,
            mem_limit=self.config.memory_limit,
            cpu_period=100000,
            cpu_quota=int(self.config.cpu_limit * 100000),
            working_dir=self.config.work_dir,
            volumes=binds,
            network_disabled=not self.config.network_enabled,
        )

        # 等待容器启动
        await self._wait_for_ready()

    async def _wait_for_ready(self, timeout: int = 30) -> None:
        """等待容器就绪"""
        start = time.time()
        while time.time() - start < timeout:
            try:
                result = await self.run_command("echo ready", timeout=5)
                if "ready" in result.lower():
                    return
            except:
                pass
            await asyncio.sleep(0.5)
        raise TimeoutError("Container failed to start within timeout")
```

### 8.4.2 命令执行

```python
async def run_command(
    self,
    command: str,
    timeout: Optional[int] = None,
) -> str:
    """在容器中执行命令"""
    if not self.container:
        raise RuntimeError("Container not created")

    timeout = timeout or self.config.timeout

    try:
        # 执行命令
        result = self.container.exec_run(
            cmd=f"bash -c '{command}'",
            workdir=self.config.work_dir,
        )

        output = result.output.decode("utf-8")

        # 检查退出码
        if result.exit_code != 0:
            raise RuntimeError(f"Command failed: {output}")

        return output

    except Exception as e:
        raise RuntimeError(f"Command execution failed: {str(e)}")
```

### 8.4.3 文件操作

```python
async def copy_from(self, container_path: str, local_path: str) -> None:
    """从容器复制文件到本地"""
    if not self.container:
        raise RuntimeError("Container not created")

    # 使用 tar 归档传输
    tar_stream, stat = self.container.get_archive(container_path)

    # 创建本地目录
    os.makedirs(os.path.dirname(local_path), exist_ok=True)

    # 写入文件
    with open(local_path, "wb") as f:
        for chunk in tar_stream:
            f.write(chunk)

async def copy_to(self, local_path: str, container_path: str) -> None:
    """从本地上传文件到容器"""
    if not self.container:
        raise RuntimeError("Container not created")

    # 创建 tar 归档
    import tarfile
    import io

    tar_stream = io.BytesIO()
    with tarfile.open(fileobj=tar_stream, mode="w") as tar:
        tar.add(local_path, arcname=os.path.basename(local_path))
    tar_stream.seek(0)

    # 上传到容器
    self.container.put_archive(os.path.dirname(container_path), tar_stream)

async def read_file(self, path: str) -> str:
    """读取容器文件内容"""
    result = await self.run_command(f"cat {path}")
    return result

async def write_file(self, path: str, content: str) -> None:
    """写入内容到容器文件"""
    # Base64 编码以处理特殊字符
    import base64
    encoded = base64.b64encode(content.encode("utf-8")).decode("utf-8")
    await self.run_command(f"echo {encoded} | base64 -d > {path}")
```

### 8.4.4 资源清理

```python
async def cleanup(self) -> None:
    """清理容器和资源"""
    if self.container:
        try:
            self.container.stop(timeout=5)
        except:
            pass
        try:
            self.container.remove(force=True)
        except:
            pass
        self.container = None

    if self.docker_client:
        self.docker_client.close()
        self.docker_client = None
```

## 8.5 SandboxManager 管理器

**文件位置**：`app/sandbox/core/manager.py`

```python
class SandboxManager:
    """沙箱生命周期管理器"""

    def __init__(self):
        self.sandboxes: Dict[str, LocalSandboxClient] = {}
        self.default_config = SandboxSettings()

    async def get_or_create(
        self,
        sandbox_id: str,
        config: Optional[SandboxSettings] = None,
    ) -> LocalSandboxClient:
        """获取或创建沙箱"""
        if sandbox_id not in self.sandboxes:
            client = LocalSandboxClient()
            await client.create(config or self.default_config)
            self.sandboxes[sandbox_id] = client

        return self.sandboxes[sandbox_id]

    async def destroy(self, sandbox_id: str) -> None:
        """销毁沙箱"""
        if sandbox_id in self.sandboxes:
            await self.sandboxes[sandbox_id].cleanup()
            del self.sandboxes[sandbox_id]

    async def destroy_all(self) -> None:
        """销毁所有沙箱"""
        for sandbox_id in list(self.sandboxes.keys()):
            await self.destroy(sandbox_id)
```

## 8.8 安全特性

1. **资源限制**
   - 内存限制：防止内存溢出
   - CPU 限制：防止资源耗尽
   - 超时控制：防止无限执行

2. **网络隔离**
   - 默认禁用网络访问
   - 可配置启用特定网络

3. **文件系统隔离**
   - Docker 容器文件系统隔离
   - 可配置卷挂载映射

4. **进程隔离**
   - 容器内进程隔离
   - 无法访问宿主机进程
