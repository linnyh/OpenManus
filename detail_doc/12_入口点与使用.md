## 12.2 main.py - Manus 主入口

**文件位置**：`main.py`

```python
import argparse
import asyncio

from app.agent.manus import Manus
from app.logger import logger


async def main():
    """主入口函数"""
    # 解析命令行参数
    parser = argparse.ArgumentParser(description="Run Manus agent with a prompt")
    parser.add_argument(
        "--prompt",
        type=str,
        required=False,
        help="Input prompt for the agent"
    )
    args = parser.parse_args()

    # 创建并初始化 Manus Agent
    agent = await Manus.create()

    try:
        # 获取用户输入
        prompt = args.prompt if args.prompt else input("Enter your prompt: ")
        if not prompt.strip():
            logger.warning("Empty prompt provided.")
            return

        logger.warning("Processing your request...")
        await agent.run(prompt)
        logger.info("Request processing completed.")

    except KeyboardInterrupt:
        logger.warning("Operation interrupted.")
    finally:
        # 确保清理资源
        await agent.cleanup()


if __name__ == "__main__":
    asyncio.run(main())
```

### 使用方式

```bash
# 交互式输入
python main.py

# 命令行参数
python main.py --prompt "帮我搜索 Python 教程"

# 带环境变量
ANTHROPIC_API_KEY=xxx python main.py
```

## 12.3 run_flow.py - Flow 工作流入口

**文件位置**：`run_flow.py`

```python
import asyncio
import time

from app.agent.data_analysis import DataAnalysis
from app.agent.manus import Manus
from app.config import config
from app.flow.flow_factory import FlowFactory, FlowType
from app.logger import logger


async def run_flow():
    """运行 PlanningFlow 工作流"""
    # 创建 Agent 字典
    agents = {
        "manus": Manus(),
    }

    # 如果启用，添加数据分析 Agent
    if config.run_flow_config.use_data_analysis_agent:
        agents["data_analysis"] = DataAnalysis()

    try:
        # 获取用户输入
        prompt = input("Enter your prompt: ")

        if prompt.strip().isspace() or not prompt:
            logger.warning("Empty prompt provided.")
            return

        # 创建 PlanningFlow
        flow = FlowFactory.create_flow(
            flow_type=FlowType.PLANNING,
            agents=agents,
        )

        logger.warning("Processing your request...")

        # 执行工作流（带超时保护）
        try:
            start_time = time.time()
            result = await asyncio.wait_for(
                flow.execute(prompt),
                timeout=3600,  # 1小时超时
            )
            elapsed_time = time.time() - start_time
            logger.info(f"Request processed in {elapsed_time:.2f} seconds")
            logger.info(result)

        except asyncio.TimeoutError:
            logger.error("Request processing timed out after 1 hour")
            logger.info("Please try a simpler request.")

    except KeyboardInterrupt:
        logger.info("Operation cancelled by user.")
    except Exception as e:
        logger.error(f"Error: {str(e)}")


if __name__ == "__main__":
    asyncio.run(run_flow())
```

### 使用方式

```bash
# 交互式输入
python run_flow.py

# 启用数据分析 Agent（需在 config.toml 中配置）
[runflow]
use_data_analysis_agent = true
```

## 12.4 run_mcp.py - MCP Agent 入口

```python
import asyncio
import argparse

from app.agent.mcp import MCPAgent
from app.logger import logger


async def main():
    """MCP Agent 入口"""
    parser = argparse.ArgumentParser(description="Run MCP Agent")
    parser.add_argument(
        "--server",
        type=str,
        required=True,
        help="MCP server connection (URL or command)"
    )
    parser.add_argument(
        "--type",
        type=str,
        choices=["sse", "stdio"],
        default="sse",
        help="Connection type"
    )
    parser.add_argument(
        "--prompt",
        type=str,
        help="Initial prompt"
    )
    args = parser.parse_args()

    # 创建 MCP Agent
    agent = MCPAgent()

    # 连接 MCP 服务器
    if args.type == "sse":
        await agent.connect_mcp_server(args.server)
    else:
        await agent.connect_mcp_server(args.server, use_stdio=True)

    try:
        prompt = args.prompt or input("Enter your prompt: ")
        await agent.run(prompt)
    finally:
        await agent.cleanup()


if __name__ == "__main__":
    asyncio.run(main())
```

### 使用方式

```bash
# SSE 连接
python run_mcp.py --server http://localhost:3000/sse --type sse

# Stdio 连接
python run_mcp.py --server "npx" --type stdio -- \
    -y @modelcontextprotocol/server-filesystem /tmp
```

## 12.5 run_mcp_server.py - MCP Server 入口

**文件位置**：`run_mcp_server.py`

```python
import argparse
import asyncio
import logging

logging.basicConfig(
    level=logging.INFO,
    handlers=[logging.StreamHandler(sys.stderr)]
)

from app.mcp.server import MCPServer


def parse_args() -> argparse.Namespace:
    """解析命令行参数"""
    parser = argparse.ArgumentParser(description="OpenManus MCP Server")
    parser.add_argument(
        "--transport",
        choices=["stdio"],
        default="stdio",
        help="Communication method"
    )
    return parser.parse_args()


if __name__ == "__main__":
    args = parse_args()

    # 创建并运行服务器
    server = MCPServer()
    server.run(transport=args.transport)
```

### 使用方式

```bash
# 启动 stdio 服务器
python run_mcp_server.py

# 作为 MCP 客户端运行
npx @anthropic/mcp-client run python run_mcp_server.py
```

### 12.6.2 自定义 Agent

```python
import asyncio
from app.agent.toolcall import ToolCallAgent
from app.tool.base import ToolCollection
from app.tool.python_execute import PythonExecute
from app.tool.str_replace_editor import StrReplaceEditor

class MyAgent(ToolCallAgent):
    name: str = "my_agent"
    description: str = "My custom agent"

        default_factory=lambda: ToolCollection(
            PythonExecute(),
            StrReplaceEditor(),
        )

async def main():
    agent = MyAgent()
    result = await agent.run("Calculate 2+2")
    print(result)

asyncio.run(main())
```

## 12.7 配置快速开始

### 1. 复制配置模板

```bash
cp config/config.example.toml config/config.toml
```

### 2. 编辑配置

```toml
[llm]
model = "claude-3-7-sonnet"
base_url = "https://api.anthropic.com/v1/"
api_key = "your-api-key-here"
```

### 3. 配置 MCP 服务器（可选）

```bash
cat > config/mcp.json << 'EOF'
{
  "mcpServers": {
    "filesystem": {
      "type": "stdio",
      "command": "npx",
      "args": ["-y", "@modelcontextprotocol/server-filesystem", "/tmp"]
    }
  }
}
EOF
```

### 4. 运行

```bash
python main.py
```
